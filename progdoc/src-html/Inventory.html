001import java.util.ArrayList;
002
003/**
004 * The Inventory Class is used to create an instance (Object) of an inventory.
005 * It is used to store multiple Item Objects together and their total price.
006 * 
007 * @author Gervaise Pierre
008 * @version Main Branch
009 */
010public class Inventory
011{
012    //Attributes
013    private ArrayList<Item> aItemList;
014    private int aTotalPrice;
015    private int aTotalVolume;
016    
017    
018    //Constructors.
019    /**
020     * Creates an Inventory Object. Item list is empty. Total price is set to 0. Total volume is set to 0.
021     */
022    public Inventory()
023    {
024        this.aItemList = new ArrayList<Item>();
025        this.setTotalPrice(0);
026        this.setTotalVolume(0);
027    }   //Inventory()
028    
029    
030    //Set Methodes. (Related to this Class)
031    /**
032     * Used to change the total price of the inventory.
033     */
034    private void setTotalPrice(final int pTotalPrice){
035        this.aTotalPrice = pTotalPrice;
036    }   //setTotalPrice()
037    
038    /**
039     * Used to change the total volume of the inventory.
040     */
041    private void setTotalVolume(final int pTotalVolume){
042        this.aTotalVolume = pTotalVolume;
043    }   //setTotalVolume()
044    
045    
046    //Get Methodes. (Related to this Class)
047    /**
048     * Used to fetch the total price of the inventory.
049     */
050    public int getTotalPrice(){
051        return this.aTotalPrice;
052    }   //getTotalPrice()
053    
054    /**
055     * Used to fetch the total volume of the inventory.
056     */
057    public int getTotalVolume(){
058        return this.aTotalVolume;
059    }   //getTotalVolume()
060    
061    /**
062     * Used to fetch an Item from the inventory.
063     */
064    public Item getItem(final String pItemName){
065        for(Item vElement : this.aItemList){
066            if(vElement.getName().toLowerCase().equals(pItemName.toLowerCase())){
067                System.out.println("True");
068                return vElement;
069            }
070        }
071        return null;
072    }   //getItem()
073    
074    
075    //Custom Methodes. (Related to this Class)
076    /**
077     * Used to check if the inventory has Item in the inventory.
078     */
079    public boolean hasItem(final String pItemName){
080        return this.getItem(pItemName) != null;
081    }   //hasItem()
082    
083    /**
084     * Used to add an item to the inventory.
085     */
086    public void addItem(final Item pItem){
087        this.aItemList.add(pItem);
088        this.setTotalPrice(this.getTotalPrice() + pItem.getPrice());
089        this.setTotalVolume(this.getTotalVolume() + pItem.getVolume());
090    }   //addItem()
091    
092    /**
093     * Used to remove an item from the inventory.
094     */
095    public void removeItem(final Item pItem){
096        String pItemName = pItem.getName();
097        if(this.hasItem(pItemName)){
098            this.setTotalPrice(this.getTotalPrice() - getItem(pItemName).getPrice());
099            this.setTotalVolume(this.getTotalVolume() - getItem(pItemName).getVolume());
100            this.aItemList.remove(pItem);
101        }
102    }   //removeItem()
103    
104    /**
105     * Returns a simple list of all the items names in the inventory for use of Display.
106     */
107    private String getItemListString(){
108        String vOutput = "";
109        for(Item vElement : this.aItemList){
110            if(vOutput != ""){
111                vOutput += ", ";
112            }
113            vOutput += vElement.getName();
114        }
115        return vOutput;
116    }   //getItemListString()
117    
118    /**
119     * Returns a table of all items inside the inventory.
120     */
121    private String[] getItemListTable(){
122        return this.getItemListString().split(", ");
123    }   //getItemListTable()
124    
125    /**
126     * Returns a detailed listing of all items inside of the inventory.
127     */
128    public String getInventoryListDescription(){
129        if(this.aItemList.isEmpty()){
130            return "- Inventory is Empty.";
131        }else{
132            String vOutput = "";
133            for(Item vElement : this.aItemList){
134                if(vOutput != ""){
135                    vOutput += "\n";
136                }
137                int vPrice = vElement.getPrice();
138                String vCred = "";
139                if(vPrice == 1){
140                    vCred = "credit";
141                }else{
142                    vCred = "credits";
143                }
144                vOutput += "- " + vElement.getName() + " : " + vPrice + " " + vCred + " | " + vElement.getVolume() + " L.";
145            }
146            return vOutput;
147        }
148    }   //getInventoryListDescription()
149    
150    
151    //Overrides. (Related to this Class)
152    /**
153     * Custom override of .toString(). ("ItemList + " : " + TotalPrice").
154     */
155    @Override public String toString()
156    {
157        if(this.getItemListString() != ""){
158            String vEndPart = "";
159            if(this.getTotalPrice() == 1){
160                vEndPart = " credit";
161            }else{
162                vEndPart = " credits";
163            }
164            return "" + this.getItemListString() + " [" + this.getTotalPrice() + vEndPart + "]";
165        }else{
166            return "";
167        }
168        
169    }
170}
