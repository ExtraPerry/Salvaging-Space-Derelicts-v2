001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * This class is part of the "Salvaging Space Derelicts" application. 
006 * Based off of "World of Zuul" which is a very simple, text based adventure game.  
007 * 
008 * This class holds an enumeration table of all command words known to the game and a description of the commands.
009 * It is used to recognise commands as they are typed in and get information about the command.
010 *
011 * @author  Michael Kolling and David J. Barnes + D.Bureau [Modified by P.Gervais]
012 * @version 2008.03.30 + 2019.09.25 [Modified]
013 */
014public class CommandWords
015{
016    // a constant array that will hold all valid command words
017    private HashMap<String, String> aValidCommands;
018
019    /**
020     * Constructor - initialise the command words.
021     */
022    public CommandWords()
023    {
024        this.aValidCommands = new HashMap<String, String>();
025        
026        this.addValidCommand("go","Moves the player from room to room, using north, east, south, west, above and below.");
027        this.addValidCommand("help","Gives the player a list of all available commands.");
028        this.addValidCommand("quit","Ends the game instance.");
029        this.addValidCommand("look","Gives the player a description of the room, it's viable exits and items inside the room.");
030        this.addValidCommand("use","Makes the player use an item.");
031        this.addValidCommand("back","Makes the player return to the previous room they were in until there is no room to return to. You can add a number after the command to go back multiple times.");
032        this.addValidCommand("take","Take an item from the room the player is in and put it in the player's inventory.");
033        this.addValidCommand("drop","Drop an item from the player's and put's in in the room's inventory the player is currently in.");
034        this.addValidCommand("inventory","Gives out a detailed list of specified inventory. [Player or Room]");
035        this.addValidCommand("test","Used to test various components of the entire game and make sure they all work properly.");
036        this.addValidCommand("merp","Easter Egg ig :D");
037    } // CommandWords()
038
039    private void addValidCommand(final String pWord, final String pDescription){
040        this.aValidCommands.put(pWord, pDescription);
041    }
042    
043    /**
044     * Check whether a given String is a valid command word. 
045     * @return true if a given string is a valid command,
046     * false if it isn't.
047     */
048    public boolean isCommand( final String pString )
049    {
050        Set<String> keys = this.aValidCommands.keySet(); //Get all keys from the HashMap
051        for (String vCommand : keys){ //"for" each key of the HashMap
052            if (vCommand.equals(pString.toLowerCase())){
053                return true;
054            }
055        } // for
056        // if we get here, the string was not found in the commands :
057        return false;
058    } // isCommand()
059    
060    /**
061     * Returns a list of all available commands. (String)
062     */
063    public HashMap getCommandHashMap()
064    {
065        return this.aValidCommands;
066    }
067    
068    /**
069     * Returns information on the command put in parameters.
070     */
071    public String getCommandInfo(final String pCommandIndexName){
072        return this.aValidCommands.get(pCommandIndexName);
073    }
074} // CommandWords
