198    /**
199     * Reads the text written by the player and translates it to a readable command format.
200     * Then sends it to be processed and run if recongnised.
201     */
202    public void interpretUITextCommand(final String pText){
203        this.aUI.println("> " + pText + ""); //Send to log the command that was read.
204        boolean vExitCondition = this.processCommand(this.aParser.getCommand(pText)); //Translate & Process the command words.
205        if(vExitCondition){
206            this.endGame();
207        }

210    /**
211     * Command manager used after the commands are filtered through the Parser class.
212     * Returns true (boolean) if the player wants to quit the game. In other words to end the instance.
213     */
214    private boolean processCommand(final Command pCommand){
215        if (pCommand.getCommandWord() != null){
216            switch(pCommand.getCommandWord().toLowerCase()){
217                case "go":
218                    goRoom(pCommand);
219                    return false;
220                case "help":
221                    printHelp(pCommand);
222                    return false;
223                case "quit":
224                    return quit(pCommand);
225                case "look":
226                    look();
227                    return false;
228                case "use":
229                    use();
230                    return false;
231                case "back":
232                    back(pCommand);
233                    return false;
234                case "take":
235                    take(pCommand);
236                    return false;
237                case "drop":
238                    drop(pCommand);
239                    return false;
240                case "inventory":
241                    inventory(pCommand);
242                    return false;
243                case "test":
244                    test(pCommand);
245                    return false;
246                case "merp":
247                    this.aUI.println("The Sergal goes Merp and the Cheese Wedge be Yellow !");
248                    return false;
249            }
250        }
251        this.aUI.println("I don't know what you mean...");
252        return false;
253    }   //processCommand()

255    /**
256     * Sends the player to the next selected room by "Command"'s order.
257     * Param = Command.
258     * return = void.
259     */
260    private void goRoom(final Command pCommand){
261        if (pCommand.hasSecondWord()){ //Checks if the go command has a second word
262            String vDirection = pCommand.getSecondWord().toLowerCase();
263            if(this.getActivePlayer().hasNextRoom(vDirection)){
264                this.aActivePlayer.moveToNextRoom(vDirection);
265                this.updateLocation();
266            }else{
267                this.aUI.println("Unknown Direction"); //Tells the player that the direction is incorrect.
268            }
269        }else{
270            this.aUI.println("Go where ?"); //If the go command has no second word asks the player where to go.
271        }
272    }   //goRoom()

274    /**
275     * Help command. //Needs to be able to adapt to all the commands available :D
276     */
277    private void printHelp(final Command pCommand){
278        if(pCommand.hasSecondWord()){
279            String vSecondWord = pCommand.getSecondWord().toLowerCase();
280            if(this.aParser.getCommandDescription(pCommand.getSecondWord().toLowerCase()) != null){
281                this.aUI.print("The command called \"" + this.formatWord(vSecondWord) + "\" is for : ");
282                this.aUI.println(this.aParser.getCommandDescription(vSecondWord));
283            }else{
284                this.aUI.println("" + this.formatWord(vSecondWord) + " is not an existing command.");
285            }
286        }else{
287            this.aUI.println("Your command words are:");
288            String[] vValidCommandList = this.aParser.getValidCommands();
289            String vOutput = "";
290            for(int i=0 ; i<vValidCommandList.length ; i++){
291                if(vOutput != ""){
292                    vOutput += ", ";
293                }
294                vOutput += vValidCommandList[i];
295            }
296            this.aUI.println("" + vOutput + ".");
297        }
298    }   //printHelp()

325    /**
326     * Back command that will send the player to the previous room or rooms.
327     */
328    private void back(final Command pCommand){
329        int vStack;
330        if(pCommand.hasSecondWord()){
331            try{
332                vStack = Integer.parseInt(pCommand.getSecondWord());
333            }catch(NumberFormatException vException){
334                String vSecondWord = pCommand.getSecondWord();
335                this.aUI.println(this.formatWord(vSecondWord) + " is not a number.");
336                return;
337            }
338            if(vStack == 0){
339                this.aUI.println("0 is not a valid number.");
340                return;
341            }
342            if(vStack < 0){
343                this.aUI.println("You cannot input a negative number.");
344                return;
345            }
346        }else{
347            vStack = 1;
348        }
349        if(this.getActivePlayer().hasPreviousRoom()){
350            this.getActivePlayer().moveToPreviousRoom(vStack);
351            this.updateLocation();
352        }else{
353            this.aUI.println("There are no rooms to go back to.");
354        }
355    }   //back()

357    /**
358     * Take command that will move an item from the room the player is in into the player's inventory.
359     */
360    private void take(final Command pCommand){
361        if(pCommand.hasSecondWord()){
362            String vItemNameCommand = pCommand.getSecondWord();
363            Room vCurrentRoom = this.getActivePlayer().getCurrentRoom();
364            if(vCurrentRoom.hasItemInInventory(vItemNameCommand)){
365                Item vItem = vCurrentRoom.getItemFromInventory(vItemNameCommand);
366                if(this.getActivePlayer().hasSpaceInInventoryFor(vItem)){
367                    this.getActivePlayer().getCurrentRoom().removeItemFromInventory(vItem);
368                    this.getActivePlayer().addItemToInventory(vItem);
369                    this.aUI.println("You've taken " + vItem.getName() + " from " + vCurrentRoom.getName());
370                }else{
371                    this.aUI.println("Not enough place in inventory to pick up " + vItem.getName() + ".");
372                }
373            }else{
374                this.aUI.println("Specified item (" + this.formatWord(vItemNameCommand) + ") cannot be found inside this room.");
375            }
376        }else{
377            this.aUI.println("Take what Item ?");
378        }
379    }   //take()
