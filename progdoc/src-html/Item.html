002/**
003 * The Item Class is used to create an instance (Object) of an item.
004 * It is used to store information about an item such as it's name, price, volume and a description of it.
005 *
006 * @author Gervaise Pierre
007 * @version Main Branch
008 */
009public class Item
010{
011    //Attributes
012    private String aName;
013    private int aPrice;
014    private int aVolume;
015    private String aItemDescription;
016    
017    
018    //Constructors
019    public Item(final String pName, final int pPrice, final int pVolume, final String pItemDescription)
020    {
021        this.aName = pName;
022        this.aPrice = pPrice;
023        this.aItemDescription = pItemDescription;
024        this.aVolume = pVolume;
025    }
026    
027    
028    //Get Methodes. (Related to this Class)
029    public String getName(){
030        return this.aName;
031    }
032    
033    public int getPrice(){
034        return this.aPrice;
035    }
036    
037    public int getVolume(){
038        return this.aVolume;
039    }
040    
041    public String getItemDescription(){
042        return this.aItemDescription;
043    }
044    
045    
046    //Custom Methodes. (Related to this Class)
047    public String getLongItemDescription(){
048        String vOutput = this.getItemDescription() + "\n" + "Price: " + this.getPrice();
049        if(this.aPrice == 1){
050            vOutput += " credit";
051        }else{
052            vOutput += " credits";
053        }
054        vOutput += ", Volume: " + this.getVolume() + " L.";
055        return vOutput;
056    }
057}
